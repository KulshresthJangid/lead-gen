3.2  deduplicator.js
──────────────────────────────────────────────────────────────────
import crypto from 'crypto';
import levenshtein from 'fast-levenshtein';

function normalizeEmail(email) {
  return email.toLowerCase().trim();
}

function hashEmail(email) {
  return crypto.createHash('sha256')
    .update(normalizeEmail(email))
    .digest('hex');
}

async function isDuplicate(db, lead) {
  // Guard: skip if email missing
  if (!lead.email) return { isDupe: true, reason: 'missing_email' };

  // Check 1: exact email hash
  const byHash = await db.get(
    'SELECT id FROM leads WHERE email_hash = ?',
    hashEmail(lead.email)
  );
  if (byHash) return { isDupe: true, reason: 'email_hash_match' };

  // Check 2: fuzzy full_name within same company_domain
  if (lead.company_domain && lead.full_name) {
    const candidates = await db.all(
      'SELECT id, full_name FROM leads WHERE company_domain = ?',
      lead.company_domain.toLowerCase().trim()
    );
    for (const c of candidates) {
      if (levenshtein.get(c.full_name, lead.full_name) < 3) {
        return { isDupe: true, reason: 'fuzzy_name_domain_match' };
      }
    }
  }
  return { isDupe: false };
}

// Filter an array of raw leads; returns { unique: [], dupes: [] }
export async function filter(db, leads) {
  const unique = [], dupes = [];
  for (const lead of leads) {
    const result = await isDuplicate(db, lead);
    if (result.isDupe) {
      dupes.push({ lead, reason: result.reason });
    } else {
      // Pre-compute hash for storage
      lead.email_hash = hashEmail(lead.email);
      unique.push(lead);
    }
  }
  return { unique, dupes };
}

RawLead shape:

──────────────────────────────────────────────────────────────────
CONSTANTS:

SYSTEM PROMPT (injected into 'prompt' field — not 'system'):
──────────────────────────────────────────────────────────
[SYSTEM]
You are a B2B lead enrichment and qualification engine.

STRICT RULES:
- Output ONLY valid JSON. No markdown. No preamble. No code fences.
- Do NOT invent, hallucinate, or modify: full_name, company_name,
  email, company_domain. Return them EXACTLY as received.
- If a field cannot be inferred, return "" (empty string).
- Enrich ONLY: pain_points, reason_for_outreach,
  lead_quality (hot|warm|cold), confidence_score (0–100).
- lead_quality = hot  → strong ICP fit, clear pain, decision-maker
- lead_quality = warm → partial fit or unclear role
- lead_quality = cold → poor fit or insufficient data

OUTPUT FORMAT (strict):
{
  "leads": [
    {
      "full_name": "",
      "job_title": "",
      "company_name": "",
      "company_domain": "",
      "email": "",
      "linkedin_url": "",
      "location": "",
      "pain_points": "",
      "reason_for_outreach": "",
      "lead_quality": "hot|warm|cold",
      "confidence_score": 0
    }
  ]
}
[/SYSTEM]
──────────────────────────────────────────────────────────

FIRST PASS — enrichBatch(leads):
  - Chunk leads into groups of BATCH_SIZE
  - Build prompt: system prompt + JSON-encoded lead array
  - POST to Ollama /api/generate with stream: false
  - Extract response.response string
  - Strip ```json ... ``` fences if present
  - JSON.parse(); on failure: retry once with explicit instruction
    "Return ONLY the JSON with no other text."
  - On second failure: log error, skip batch, return raw leads
  - Merge enriched fields back onto original lead objects

SECOND PASS — refineOutreach(leads):
  - Process one lead at a time
  - Prompt addition:
    "Improve ONLY the reason_for_outreach field.
     Make it hyper-personalized (reference their company/role/domain),
     under 3 sentences, no generic sales language.
     Return complete lead JSON (single object, not array)."
  - Same parse-retry-skip logic
  - Update only reason_for_outreach on the lead object

OFFLINE RESILIENCE:
  - Wrap all Ollama calls in try/catch
  - If Ollama unreachable: set enriched_at=null, enrichment_attempts+=1
  - Return leads with AI fields empty — pipeline continues without enrichment
  - Emit socket event 'ollama_offline' with timestamp

3.5  scheduler.js
──────────────────────────────────────────────────────────────────
- Read scraping_interval from config.json on startup
  (valid values: 15, 30, 60, 360 — in minutes)
- Build cron expression dynamically: `*/${interval} * * * *`
- Maintain pipelineState object:
    { lastRunAt, nextRunAt, status, lastRunStats }
- Pipeline run sequence:
    1. Generate run_id (uuid v4), insert pipeline_log row (status=running)
    2. Load scraper_targets from config.json
    3. raw = await scraper.scrapeLeads(targets)
    4. { unique, dupes } = await deduplicator.filter(db, raw)
    5. enriched = await enricher.enrichBatch(db, unique)
    6. refined = await enricher.refineOutreach(enriched)
    7. inserted = await db.insertLeads(refined)
    8. Update pipeline_log: status=success, counts, finished_at
    9. io.emit('new_leads', { count: inserted.length, runId })
   10. Update pipelineState

- Expose: { getState(), triggerNow(), reschedule(intervalMinutes) }
- Log each step with pino logger: [PIPELINE][run_id] Step N complete

6. SECURITY & RELIABILITY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

- All API inputs validated with zod before DB operations
- express-rate-limit on all routes: 100 req/15min per IP
- Parameterized queries only — no string interpolation in SQL
- Ollama response validated with JSON.parse() in try/catch — retry once, then discard
- Scraper never stores a lead without a valid email (contains '@' and '.')
- config.json never exposes DATABASE_URL (use env var only)
- CORS restricted to localhost:3000 in dev; configurable ALLOWED_ORIGINS in .env
- Content-Security-Policy headers via helmet on backend
- No external AI API keys stored — Ollama is local-only
- CSV export sanitizes cell values to prevent CSV injection
  (prefix cells starting with =, +, -, @ with a single quote)

7. ERROR HANDLING & RESILIENCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

- Pipeline never crashes the server: all steps wrapped in try/catch
- Each step failure logged to pipeline_log.errors_json
- Partial success: if enrichment fails, raw scraped leads still saved
- DB write failures: retry 3x with exponential backoff (100ms, 200ms, 400ms)
- If pipeline is already running (from cron + manual trigger race):
  skip the new run, log a warning, return 409 from POST /api/pipeline/trigger
- Graceful shutdown: SIGTERM handler waits for in-progress pipeline run before exit
- Frontend: axios interceptor catches 429/503 → shows "Server busy" toast
